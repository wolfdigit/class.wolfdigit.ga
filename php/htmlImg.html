#include <iostream>
using namespace std;

int main() {
    int n, i, count=0;

    cout << "輸入一正整數，列出此數的所有正因數，並判斷此數字是否為質數" << endl;
    cout << "請輸入一個數字:";
    cin >> n;

    cout << n << "的因數有:";
    for (i=1; i<n; i+=1) {
        if (n%i==0) {
            count += 1;
            cout << i << " ";
        }
    }
    count += 1;
    cout << n << endl;

    if (count==2) {
        cout << n << "為質數" << endl;
    }
    else {
        cout << n << "不為質數" << endl;
    }

    return 0;
}
